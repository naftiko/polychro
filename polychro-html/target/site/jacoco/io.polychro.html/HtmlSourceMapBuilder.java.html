package io.polychro.html;

import io.polychro.spi.SourceMap;
import io.polychro.spi.SourceRange;

import java.util.LinkedHashMap;
import java.util.Map;

/**
 * Builds a path-to-range source map for projected HTML nodes.
 */
class HtmlSourceMapBuilder {

    static final String PRECISION_EXACT = "exact";
    static final String PRECISION_APPROXIMATE = "approximate";

    private final Map<String, SourceRange> ranges = new LinkedHashMap<>();
    private final Map<String, String> precisions = new LinkedHashMap<>();

    void put(String path, SourceRange range, String precision) {
        if (path == null || range == null) {
            return;
        }
        ranges.put(path, range);
        precisions.put(path, precision == null ? PRECISION_APPROXIMATE : precision);
    }

    String precisionOf(String path) {
        return precisions.get(path);
    }

    SourceMap build() {
        Map<String, SourceRange> snapshot = Map.copyOf(ranges);
        return snapshot::get;
    }
}
